

## 1. List

### List คืออะไรและทำงานอย่างไร?
* **List** คือโครงสร้างข้อมูลที่ใช้สำหรับเก็บสมาชิก (Elements) หลายๆ ตัวที่มีชนิดข้อมูล (Data type) เดียวกัน 
* มีขนาดที่สามารถปรับเปลี่ยนให้ใหญ่ขึ้นหรือเล็กลงได้แบบอัตโนมัติ (Dynamically grow in size) 
* คุณสามารถเพิ่ม (Add) และลบ (Remove) ข้อมูลได้ทั้งจากส่วนหน้าสุด (Beginning) และส่วนท้ายสุด (End) ของ List 
* การเข้าถึงข้อมูลจะเป็นแบบ **Sequential access** หรือเข้าถึงตามลำดับเท่านั้น 

### ข้อดี - ข้อเสีย ของ List
* **ข้อดี:** ทำงานได้รวดเร็วมากในส่วนของการแทรก (Insert) และการลบ (Delete) ข้อมูล เมื่อนำไปเปรียบเทียบกับ Arrays และ Vectors 
* **ข้อเสีย:** ไม่รองรับการเข้าถึงข้อมูลแบบสุ่ม (No random access) ซึ่งหมายความว่าคุณไม่สามารถกระโดดข้ามไปยังตำแหน่งที่ต้องการ หรือเข้าถึงข้อมูลด้วยตัวเลขดัชนี (Index) ตรงๆ ได้ 

### ตัวอย่างโค้ดการใช้งาน List และคำอธิบาย

```cpp
#include <iostream>
#include <list>      // ต้องทำการ include list เพื่อใช้งาน 
#include <iterator>  // สำหรับการใช้งาน Iterator 

using namespace std;

int main() {
    // 1. การสร้าง List (Create a List)
    list<string> cars = {"Volvo", "BMW", "Ford", "Mazda"}; // สร้าง list ที่เก็บข้อมูลประเภท string 

    // 2. การเข้าถึงและการเปลี่ยนแปลงข้อมูล
    cout << "First: " << cars.front() << "\n"; // เข้าถึงตัวแรก 
    cout << "Last: " << cars.back() << "\n";   // เข้าถึงตัวสุดท้าย 
    
    cars.front() = "Opel";   // เปลี่ยนแปลงค่าข้อมูลตัวแรก 
    cars.back() = "Toyota";  // เปลี่ยนแปลงค่าข้อมูลตัวสุดท้าย 

    // 3. การเพิ่มข้อมูล (Add Elements)
    cars.push_front("Tesla"); // เพิ่ม "Tesla" ไปที่ด้านหน้าสุด 
    cars.push_back("VW");     // เพิ่ม "VW" ไปที่ด้านหลังสุด 

    // 4. การตรวจสอบขนาดและความว่างเปล่า
    cout << "Size: " << cars.size() << "\n";     // ดูขนาดของ list 
    cout << "Is empty?: " << cars.empty() << "\n"; // เช็คว่าว่างหรือไม่ (0=ไม่ว่าง, 1=ว่าง) 

    // 5. การลบข้อมูล (Remove Elements)
    cars.pop_front(); // ลบข้อมูลตัวหน้าสุดออก 
    cars.pop_back();  // ลบข้อมูลตัวหลังสุดออก 

    // 6. การใช้ Iterator แทรกและลบข้อมูล ณ ตำแหน่งที่ระบุ 
    list<string>::iterator it = cars.begin(); // ให้ Iterator ชี้ไปที่ตัวแรก 
    it++; // เลื่อนตำแหน่งไป 1 ช่อง 
    cars.insert(it, "Honda"); // แทรกข้อมูล "Honda" ณ ตำแหน่งที่ชี้อยู่ 
    cars.erase(it); // ลบข้อมูล ณ ตำแหน่งที่ Iterator ชี้อยู่ทิ้งไป 

    // 7. การลูปผ่าน List (Loop Through a List) ด้วย for-each 
    for (string car : cars) { 
        cout << car << " ";
    }
    
    return 0;
}
```

# 2. Vector

## Vector คืออะไรและทำงานอย่างไร?
* **Vector** คือโครงสร้างข้อมูลที่ใช้สำหรับจัดเก็บสมาชิกหลายๆ ตัวที่มีชนิดข้อมูล (Data type) เดียวกันเอาไว้ด้วยกัน 
* Vector มีลักษณะการทำงานเป็น **Dynamic Array** (อาร์เรย์แบบพลวัต) ซึ่งมีความสามารถพิเศษในการปรับขนาด (resize) ตัวเองได้แบบอัตโนมัติเมื่อมีการแทรก (insert) หรือลบ (delete) ข้อมูล 
* เป็นส่วนหนึ่งของ Standard Template Library (STL) ของภาษา C++ และมีฟังก์ชันที่เป็นประโยชน์มากมายให้เรียกใช้สำหรับการจัดการข้อมูล 

## ข้อดี - ข้อเสีย ของ Vector
* **ข้อดี:** * รองรับการเข้าถึงข้อมูลแบบสุ่ม (Random Access) ทำให้สามารถเข้าถึงข้อมูลในตำแหน่ง (Index) ใดๆ ก็ได้ทันที (เช่น `v[3]`) ซึ่ง List ทำไม่ได้
  * การวนลูปอ่านข้อมูล (Traversing) ทำได้ง่ายและรวดเร็ว
* **ข้อเสีย:** * การแทรก (Insert) หรือการลบ (Delete) ข้อมูลตรงกลางหรือด้านหน้าของ Vector จะทำงานได้ช้ากว่า List  เนื่องจากต้องมีการเลื่อนตำแหน่ง (Shift) ข้อมูลตัวอื่นๆ เพื่อจัดสรรพื้นที่ใหม่

## การดำเนินการพื้นฐาน (Basic Vector Operations)
การจัดการ Vector มีการดำเนินการหลักๆ ดังนี้:
1. การเข้าถึงข้อมูล (Accessing Elements)
2. การอัปเดตข้อมูล (Updating Elements)
3. การวนลูปดูข้อมูล (Traversing Vector)
4. การเพิ่ม/แทรกข้อมูล (Inserting Elements)
5. การลบข้อมูล (Deleting Elements)

---

## ตัวอย่างโค้ดการใช้งาน Vector และคำอธิบาย

```cpp
#include <bits/stdc++.h> // โหลด header ไฟล์นี้เพื่อใช้งาน Vector และฟังก์ชัน STL ทั้งหมด 
using namespace std;

int main() {
    // ==========================================
    // 1. การสร้าง Vector (Creating Vectors)
    // ==========================================
    vector<int> v1;                       // สร้าง Vector ว่างๆ 
    vector<int> v2 = {1, 4, 2, 3, 5};     // สร้างพร้อมกำหนดค่าเริ่มต้น 
    vector<int> v3(5, 9);                 // สร้าง Vector ขนาด 5 ช่อง โดยให้ทุกช่องมีค่าเป็น 9 

    // ==========================================
    // 2. การตรวจสอบว่าว่างหรือไม่ (Checking if Empty)
    // ==========================================
    if (v1.empty()) {                     // ฟังก์ชัน .empty() คืนค่าความจริงถ้า Vector ว่าง 
        cout << "v1 is Empty\n"; 
    }
    // หมายเหตุ: สามารถเช็คได้อีกวิธีคือ if (v1.begin() == v1.end()) 

    // ==========================================
    // 3. การเข้าถึงและการอัปเดตข้อมูล (Accessing & Updating)
    // ==========================================
    cout << "Index 3: " << v2[3] << "\n";      // เข้าถึงข้อมูลด้วยการใช้ Index เหมือน Array 
    cout << "at(2): " << v2.at(2) << "\n";     // เข้าถึงข้อมูลด้วยฟังก์ชัน .at() 

    v2[3] = 99;                                // เปลี่ยนแปลงค่าใน index 3 
    v2.at(2) = 88;                             // เปลี่ยนแปลงค่าใน index 2 

    // ==========================================
    // 4. การเพิ่มและแทรกข้อมูล (Inserting Elements)
    // ==========================================
    v2.push_back(6);                           // เพิ่มค่า 6 ต่อท้าย Vector 
    v2.insert(v2.begin() + 1, 7);              // แทรกค่า 7 ลงไปที่ตำแหน่ง index 1 (ต้องใช้ iterator ระบุตำแหน่ง) 

    // ==========================================
    // 5. การลบข้อมูล (Deleting Elements)
    // ==========================================
    v2.pop_back();                             // ลบข้อมูลตัวสุดท้ายออกไป 
    v2.erase(find(v2.begin(), v2.end(), 88));  // ค้นหาตัวที่มีค่า 88 แล้วลบออก (ลบด้วย iterator) 

    // ==========================================
    // 6. การปรับขนาด และ การเรียงลำดับ (Resizing & Sorting)
    // ==========================================
    v2.resize(9);                              // ปรับขนาดของ Vector ให้มี 9 ช่อง 
    
    sort(v2.begin(), v2.end());                // เรียงลำดับข้อมูลจากน้อยไปมาก 
    // sort(v2.begin(), v2.end(), greater<>()); // หากต้องการเรียงจากมากไปน้อยให้เติม greater<>() 

    // ==========================================
    // 7. การลูปดูข้อมูล (Traversing)
    // ==========================================
    cout << "Traversing with standard loop: ";
    for (int i = 0; i < v2.size(); i++) {      // วนลูปผ่านขนาดของ Vector ด้วย v.size() 
        cout << v2[i] << " "; 
    }
    cout << "\n";

    // การวนลูปด้วย Iterator 
    cout << "Traversing with Iterator: ";
    vector<int>::iterator first = v2.begin();  // กำหนด iterator ชี้ไปที่จุดเริ่มต้น 
    vector<int>::iterator last = v2.end();     // กำหนด iterator ชี้ไปที่จุดสิ้นสุด 
    
    while(first != last) {                     // ทำซ้ำตราบใดที่ยังไม่ถึงจุดสิ้นสุด 
        cout << *first << " "; 
        first++; 
    }
    
    return 0; 
}
```

# 3. Recursion

## Recursion คืออะไร?
**Recursion** หรือ การเวียนเกิด คือ **"ฟังก์ชันที่ทำการเรียกใช้งานตัวเอง"** (A function calling itself) 
หลักการทำงานของมันคือการพยายามแก้ปัญหาใหญ่ๆ ด้วยการแบ่งปัญหาเหล่านั้นออกเป็นปัญหาย่อยๆ ที่มีรูปแบบเหมือนเดิม ทำซ้ำไปเรื่อยๆ จนกว่าปัญหานั้นจะเล็กพอที่จะหาคำตอบได้ทันทีโดยไม่ต้องแบ่งย่อยอีก

## กฎเหล็ก 2 ข้อของ Recursion
การเขียนฟังก์ชัน Recursive ที่ถูกต้องและไม่ทำให้โปรแกรมพัง (เช่น เกิด Error Stack Overflow) **ต้องประกอบด้วย 2 ส่วนนี้เสมอ**:

1. **Base Case (จุดสิ้นสุด/เงื่อนไขหยุด):** คือเงื่อนไขที่บอกฟังก์ชันว่า "หยุดเรียกตัวเองได้แล้ว และให้คืนค่า (Return) กลับไป" ถ้าไม่มีส่วนนี้ โปรแกรมจะวนลูปเรียกตัวเองไปเรื่อยๆ ไม่มีวันจบ
2. **Recursive Case (ส่วนเรียกตัวเอง):** คือส่วนที่ฟังก์ชันทำการเรียกตัวเองซ้ำ แต่มีข้อแม้ว่า **ข้อมูลที่ส่งเข้าไปประมวลผลในรอบใหม่จะต้องมีขนาดเล็กลง** หรือขยับเข้าใกล้เงื่อนไข Base Case เสมอ

---

## ตัวอย่างคลาสสิก: การหาค่า Factorial ($n!$)
สมมติเราต้องการหาค่า $5!$ ซึ่งมีค่าเท่ากับ $5 \times 4 \times 3 \times 2 \times 1$
ในทางคณิตศาสตร์ เราสามารถมอง $5!$ เป็น $5 \times 4!$ ได้ นี่คือจุดเริ่มต้นของการมองปัญหาแบบ Recursion

### ตัวอย่างโค้ด C++
```cpp
#include <iostream>
using namespace std;

// ฟังก์ชันหาค่า Factorial
int factorial(int n) {
    // 1. Base Case: จุดสิ้นสุด (เรากำหนดว่า 0! มีค่าเท่ากับ 1)
    if (n == 0) {
        return 1;
    }
    // 2. Recursive Case: เรียกตัวเองโดยลดค่า n ลงทีละ 1 เพื่อเข้าใกล้ Base Case
    else {
        return n * factorial(n - 1); 
    }
}

int main() {
    int num = 5;
    cout << "Factorial of " << num << " is " << factorial(num) << "\n";
    // ผลลัพธ์ที่ได้: Factorial of 5 is 120
    return 0;
}
```

# 4. Search Algorithms

การค้นหาข้อมูล (Searching) คือกระบวนการหาตำแหน่งของข้อมูลเป้าหมาย (Target) ภายในโครงสร้างข้อมูล (เช่น Array หรือ Vector) โดยอัลกอริทึมพื้นฐานที่สำคัญในวิชาโครงสร้างข้อมูลมี 2 รูปแบบ ได้แก่ Linear Search และ Binary Search

---

## 1. Linear Search 

**Linear Search** เป็นวิธีที่เรียบง่ายและตรงไปตรงมาที่สุด หลักการคือการเช็คข้อมูลไปทีละตัวตั้งแต่ช่องแรกไปจนถึงช่องสุดท้าย หากพบข้อมูลที่ตรงกับเป้าหมายก็จะคืนค่าตำแหน่ง (Index) นั้นกลับมา แต่ถ้าค้นหาจนจบแล้วไม่พบก็จะถือว่าไม่มีข้อมูลนั้นอยู่

### จุดเด่น - จุดด้อย
* **ข้อดี:** ทำงานได้ทันทีโดยไม่ต้องเรียงลำดับข้อมูลก่อน (Unsorted data) และเขียนโค้ดได้ง่ายมาก
* **ข้อเสีย:** ทำงานช้ามากเมื่อต้องจัดการกับข้อมูลปริมาณมหาศาล เนื่องจากในกรณีที่แย่ที่สุด (Worst-case) โปรแกรมจะต้องเช็คข้อมูลทุกตัว (ความซับซ้อนของเวลาคือ $O(n)$)

### ตัวอย่างโค้ด C++ (Linear Search)
```cpp
#include <iostream>
#include <vector>
using namespace std;

// ฟังก์ชัน Linear Search
int linearSearch(vector<int>& arr, int target) {
    // วนลูปเช็คข้อมูลทีละช่องตั้งแต่ index 0
    for (int i = 0; i < arr.size(); i++) {
        if (arr[i] == target) {
            return i; // ถ้าเจอเป้าหมาย ให้คืนค่าตำแหน่ง index
        }
    }
    return -1; // ถ้าหาจนจบแล้วไม่เจอ ให้คืนค่า -1
}

int main() {
    vector<int> numbers = {15, 3, 9, 20, 7};
    int target = 9;
    
    int result = linearSearch(numbers, target);
    
    if (result != -1) cout << "Found at index: " << result << "\n";
    else cout << "Not found\n";
    
    return 0;
}
```

## 2. Binary Search (การค้นหาแบบทวิภาค)

**Binary Search** เป็นอัลกอริทึมที่มีประสิทธิภาพสูงมาก โดยใช้หลักการ Divide and Conquer (แบ่งแยกและเอาชนะ) แต่มีเงื่อนไขบังคับคือ ข้อมูลจะต้องถูกเรียงลำดับมาแล้วเท่านั้น (Sorted data)

### หลักการทำงานคือการหาค่าที่อยู่ "ตรงกลาง" ของชุดข้อมูล แล้วนำเป้าหมายไปเปรียบเทียบ:

1. หากค่าตรงกลางคือเป้าหมาย แสดงว่าค้นหาสำเร็จ

2. หากเป้าหมาย น้อยกว่า ค่าตรงกลาง โปรแกรมจะตัดข้อมูลครึ่งขวาทิ้ง และไปค้นหาต่อในครึ่งซ้าย

3. หากเป้าหมาย มากกว่า ค่าตรงกลาง โปรแกรมจะตัดข้อมูลครึ่งซ้ายทิ้ง และไปค้นหาต่อในครึ่งขวา
   
ทำเช่นนี้ซ้ำไปเรื่อยๆ จนกว่าจะเจอเป้าหมายหรือขอบเขตการค้นหาทับซ้อนกัน

## จุดเด่น - จุดด้อย

**ข้อดี**: ค้นหาได้อย่างรวดเร็วมากแม้ข้อมูลจะมีเป็นล้านตัวก็ใช้การเปรียบเทียบเพียงไม่กี่ครั้ง **(ความซับซ้อนของเวลาคือ $O(\log n)$)**
**ข้อเสีย**: ข้อมูลต้องถูกจัดเรียง (Sort) ก่อนเสมอ ซึ่งกระบวนการจัดเรียงอาจกินเวลาเพิ่มเติม

### ตัวอย่างโค้ด C++ (Binary Search)
```cpp
#include <iostream>
#include <vector>
using namespace std;

// ฟังก์ชัน Binary Search (ข้อมูลใน Vector ต้องเรียงลำดับแล้ว)
int binarySearch(vector<int>& arr, int target) {
    int left = 0;                  // กำหนดขอบเขตซ้ายสุด
    int right = arr.size() - 1;    // กำหนดขอบเขตขวาสุด

    while (left <= right) {
        // หาตำแหน่งกึ่งกลาง (ป้องกันการเกิด overflow)
        int mid = left + (right - left) / 2; 

        if (arr[mid] == target) {
            return mid; // เจอข้อมูลที่ตำแหน่งตรงกลาง
        }
        else if (arr[mid] < target) {
            left = mid + 1; // เป้าหมายมากกว่า ตัดครึ่งซ้ายทิ้ง ขยับขอบซ้ายมาแทน
        }
        else {
            right = mid - 1; // เป้าหมายน้อยกว่า ตัดครึ่งขวาทิ้ง ขยับขอบขวามาแทน
        }
    }
    return -1; // หาไม่เจอ
}

int main() {
    // ข้อมูลเรียงลำดับจากน้อยไปมากแล้ว
    vector<int> numbers = {3, 7, 9, 15, 20, 25, 30}; 
    int target = 25;
    
    int result = binarySearch(numbers, target);
    
    if (result != -1) cout << "Found at index: " << result << "\n";
    else cout << "Not found\n";
    
    return 0;
}
```

# 5. Linked List

## Linked List คืออะไร?
**Linked List** เป็นโครงสร้างข้อมูลเชิงเส้น (Linear Data Structure) ที่ใช้สำหรับเก็บข้อมูลประเภทเดียวกันหลายๆ ตัว  โดยมีจุดเด่นคือขนาดของมันสามารถขยายหรือหดตัวได้แบบไดนามิก (Dynamically grow in size) 

แตกต่างจาก Array (หรือ Vector) ที่เก็บข้อมูลติดกันเป็นพรืดในหน่วยความจำ Linked List จะเก็บข้อมูลแยกเป็นก้อนๆ ที่เราเรียกว่า **"โหนด" (Node)** ซึ่งแต่ละโหนดจะอยู่กระจัดกระจายกันในหน่วยความจำ และใช้ **"ตัวชี้" (Pointer)** ในการเชื่อมโยงหากันเหมือนโซ่

### องค์ประกอบของ Node
ภายในโหนด 1 ตัว จะประกอบไปด้วย 2 ส่วนหลักๆ คือ:
1. **Data:** ส่วนที่ใช้เก็บข้อมูลจริง (เช่น ตัวเลข, ข้อความ)
2. **Next Pointer:** ตัวชี้ที่เก็บที่อยู่ (Address) ของโหนดถัดไปในสายโซ่

---

## ประเภทของ Linked List
1. **Singly Linked List (ลิงก์ลิสต์เดี่ยว):** แต่ละโหนดชี้ไปหาโหนดถัดไปทางเดียว (จากหัวไปหาง)
2. **Doubly Linked List (ลิงก์ลิสต์คู่):** แต่ละโหนดมีตัวชี้ 2 ตัว คือชี้ไปโหนดถัดไป (Next) และชี้กลับไปโหนดก่อนหน้า (Prev) ทำให้เดินหน้าและถอยหลังได้ (ซึ่ง `std::list` ใน C++ STL ทำงานแบบนี้)
3. **Circular Linked List (ลิงก์ลิสต์แบบวงกลม):** โหนดตัวสุดท้ายจะชี้วนกลับมาหาโหนดตัวแรกสุด ทำให้ไม่มีจุดสิ้นสุด (NULL)

---

## ข้อดี - ข้อเสีย ของ Linked List

### ข้อดี
* **ขนาดปรับเปลี่ยนได้อิสระ:** ไม่ต้องจองขนาดพื้นที่ล่วงหน้าเหมือน Array 
* **เพิ่ม/ลบ ข้อมูลได้รวดเร็ว:** การแทรก (Insert) และลบ (Delete) ข้อมูลทำได้เร็วกว่า Array และ Vector มาก  เพราะไม่ต้องเลื่อน (Shift) ตำแหน่งข้อมูลตัวอื่นๆ แค่สลับสายตัวชี้ (Pointer) ก็เสร็จแล้ว

### ข้อเสีย
* **ไม่มี Random Access:** การเข้าถึงข้อมูลต้องทำแบบเรียงลำดับ (Sequential access) เท่านั้น  หากต้องการข้อมูลตัวที่ 10 ก็ต้องเริ่มเดินจากตัวที่ 1 ไล่ไปเรื่อยๆ จนถึงตัวที่ 10 ทำให้ค้นหาข้อมูลได้ช้า
* **เปลืองหน่วยความจำเพิ่ม:** ต้องเสียพื้นที่หน่วยความจำเพิ่มเติมในทุกๆ โหนดเพื่อเก็บค่า Pointer

---

## ตัวอย่างโค้ด C++ (การสร้าง Singly Linked List เบื้องต้น)


```cpp
#include <iostream>
using namespace std;

// 1. สร้างโครงสร้างของ Node
struct Node {
    int data;       // ส่วนเก็บข้อมูล
    Node* next;     // ส่วนเก็บ Pointer ชี้ไปโหนดถัดไป
};

int main() {
    // 2. สร้างโหนด 3 ตัวในหน่วยความจำ (Dynamic Allocation)
    Node* head = new Node();
    Node* second = new Node();
    Node* third = new Node();

    // 3. กำหนดค่าข้อมูล (Data) และเชื่อมโยง (Link) โหนดเข้าด้วยกัน
    head->data = 10;        // กำหนดข้อมูลให้หัว
    head->next = second;    // ให้หัวชี้ไปหาโหนดที่ 2

    second->data = 20;      // กำหนดข้อมูลให้โหนดที่ 2
    second->next = third;   // ให้โหนดที่ 2 ชี้ไปหาโหนดที่ 3

    third->data = 30;       // กำหนดข้อมูลให้โหนดที่ 3
    third->next = nullptr;  // ตัวสุดท้ายไม่มีตัวถัดไป ให้ชี้ไปที่ NULL (nullptr)

    // 4. การวนลูปเพื่อเข้าถึงและพิมพ์ข้อมูล (Traversing)
    Node* current = head;           // เริ่มต้นจากหัว
    while (current != nullptr) {    // ทำไปเรื่อยๆ จนกว่าจะเจอ NULL
        cout << current->data << " -> ";
        current = current->next;    // กระโดดไปยังโหนดถัดไป
    }
    cout << "NULL\n";

    return 0;
}
```

# 6. Tree

![Treeimage](https://media.geeksforgeeks.org/wp-content/uploads/20251006175023711337/1.webp)

## Tree คืออะไร?
**Tree** คือโครงสร้างข้อมูลแบบไม่เป็นเชิงเส้น (Non-linear Data Structure) ที่ใช้เก็บข้อมูลในลักษณะ **"ลำดับชั้น" (Hierarchical)** โดยจำลองรูปแบบมาจากต้นไม้ที่กลับหัวลง ประกอบไปด้วยโหนด (Node) ที่เชื่อมต่อกันด้วยเส้นเชื่อม (Edge) 

โครงสร้างแบบนี้พบได้ทั่วไปในชีวิตประจำวันและในระบบคอมพิวเตอร์ เช่น ระบบจัดการไฟล์ (File System) ในคอมพิวเตอร์ที่มีโฟลเดอร์หลัก และมีโฟลเดอร์ย่อยๆ ซ้อนอยู่ข้างใน

### คำศัพท์ที่สำคัญใน Tree
1. **Root (ราก):** โหนดแรกสุดของ Tree (อยู่บนสุด) ซึ่งไม่มีโหนดไหนชี้มาที่มัน
2. **Node (โหนด):** จุดที่ใช้เก็บข้อมูล 
3. **Edge (เส้นเชื่อม):** เส้นที่เชื่อมระหว่างโหนด
4. **Parent (โหนดพ่อแม่):** โหนดที่อยู่สูงกว่าและมีเส้นเชื่อมไปยังโหนดที่อยู่ต่ำกว่า
5. **Child (โหนดลูก):** โหนดที่อยู่ต่ำกว่าและแตกแขนงออกมาจาก Parent
6. **Leaf (ใบ):** โหนดที่อยู่ปลายสุด ไม่มี Child แตกออกไปอีกแล้ว (โหนดลูกเป็น NULL)

---

## ทำไมถึงต้องใช้ Tree? (ข้อดี - ข้อเสีย)

### ข้อดี
* **สะท้อนรูปแบบข้อมูลจริง:** เหมาะสำหรับการเก็บข้อมูลที่มีโครงสร้างเป็นลำดับชั้นหรือจัดหมวดหมู่ (เช่น แผนผังองค์กร, โครงสร้าง HTML/XML)
* **การค้นหารวดเร็ว:** หากใช้ Tree ชนิดพิเศษอย่าง **Binary Search Tree (BST)** จะสามารถค้นหา, แทรก, และลบข้อมูลได้อย่างรวดเร็วมาก โดยใช้เวลาเพียง $O(\log n)$ ซึ่งเร็วกว่า Linked List หรือ Array ทั่วไป
* **ยืดหยุ่น:** ขนาดปรับเปลี่ยนได้แบบไดนามิกคล้ายกับ Linked List

### ข้อเสีย
* **ซับซ้อนกว่า:** การเขียนโค้ดเพื่อสร้างและจัดการ Tree (เช่น การลบโหนดตรงกลาง) มีความซับซ้อนกว่าโครงสร้างข้อมูลแบบเชิงเส้นมาก
* **เปลืองหน่วยความจำ:** แต่ละโหนดต้องใช้หน่วยความจำเพิ่มขึ้นเพื่อเก็บตัวชี้ (Pointer) ที่ชี้ไปยังโหนดลูกๆ

## Binary Tree (ต้นไม้ทวิภาค)


![Image of Binary Search Tree](https://media.geeksforgeeks.org/wp-content/uploads/20250904151404492797/bst1.webp)


ประเภทของ Tree ที่ถูกนำมาใช้และออกสอบบ่อยที่สุดคือ **Binary Tree** ซึ่งมีกฎง่ายๆ ว่า **"โหนดพ่อแม่ 1 ตัว จะมีโหนดลูกได้สูงสุดไม่เกิน 2 ตัว"** (เรียกว่าลูกซ้าย และลูกขวา)

### ตัวอย่างโค้ด C++ (การสร้างโหนดของ Binary Tree)
โครงสร้างโค้ดจะคล้ายกับ Linked List แต่เปลี่ยนจาก Pointer ที่ชี้ไปตัวถัดไป (next) เป็นการชี้ไปที่ลูกซ้าย (left) และลูกขวา (right) แทนครับ

```cpp
#include <iostream>
using namespace std;

// 1. สร้างโครงสร้างของ Node สำหรับ Binary Tree
struct TreeNode {
    int data;           // ส่วนเก็บข้อมูล
    TreeNode* left;     // Pointer ชี้ไปหาลูกทางซ้าย
    TreeNode* right;    // Pointer ชี้ไปหาลูกทางขวา

    // Constructor สำหรับสร้างโหนดใหม่
    TreeNode(int val) {
        data = val;
        left = nullptr;
        right = nullptr;
    }
};

int main() {
    // 2. สร้างโหนดต่างๆ ในหน่วยความจำ
    TreeNode* root = new TreeNode(10);      // โหนด Root
    TreeNode* node1 = new TreeNode(20);     // โหนดลูก
    TreeNode* node2 = new TreeNode(30);     // โหนดลูก

    // 3. เชื่อมโยงโหนดเข้าด้วยกัน
    root->left = node1;   // ให้ 20 เป็นลูกซ้ายของ 10
    root->right = node2;  // ให้ 30 เป็นลูกขวาของ 10

    // โครงสร้างที่ได้:
    //       10
    //      /  \
    //    20    30

    cout << "Root: " << root->data << "\n";
    cout << "Left Child: " << root->left->data << "\n";
    cout << "Right Child: " << root->right->data << "\n";

    return 0;
}
```

## 7. Sorting Algorithm

### 1\. Basic Sort

*   **Sorting Algorithm**: คืออัลกอริทึมที่ใช้จัดเรียงองค์ประกอบในอาเรย์หรือรายการตามลำดับที่กำหนด
    
*   **การแบ่งประเภทตามหน่วยความจำ**:
    
    *   **Internal Sort**: ข้อมูลทั้งหมดถูกเก็บไว้ในหน่วยความจำหลัก (RAM) ระหว่างกระบวนการเรียงลำดับ
        
    *   **External Sort**: ใช้หน่วยความจำหลักสำหรับข้อมูลที่กำลังเรียง และใช้หน่วยความจำสำรองสำหรับข้อมูลส่วนเกินที่ไม่พอเก็บใน RAM
        
*   **Sort Stability**: คุณสมบัติที่ข้อมูลซึ่งมีค่าเท่ากันจะยังคงลำดับเดิมตามอินพุตในผลลัพธ์ที่เรียงแล้ว
    
*   **Pass**: การกวาดอ่านข้อมูลหนึ่งรอบในระหว่างการเรียงลำดับ
    
*   **ความสำคัญ**: ช่วยลดความซับซ้อนของปัญหา และทำให้การค้นหาข้อมูลด้วยวิธี Binary Search ทำได้
    

### 2\. Another Sort Algorithm


#### 2.1 Selection Sort

*   **วิธีการ**: เลือกองค์ประกอบที่น้อยที่สุด (หรือมากที่สุด) จากส่วนที่ยังไม่เรียง แล้วสลับกับตำแหน่งแรกของส่วนนั้น ทำซ้ำไปเรื่อยๆ
    
*   **ความซับซ้อน**: $O(n^{2})$ เพราะมี Loop ซ้อนกันเพื่อเลือกและเปรียบเทียบ
    
*   **ข้อดี**: เข้าใจง่าย และใช้จำนวนการสลับ (Swap) น้อย
    
*   **ข้อเสีย**: ไม่มีความคงที่ (Not stable)
    

#### 2.2 Bubble Sort

*   **วิธีการ**: สลับองค์ประกอบที่อยู่ติดกันซ้ำๆ หากอยู่ในลำดับที่ผิด จนกว่าข้อมูลจะเรียงกันถูกต้อง
    
*   **คุณสมบัติ**: เป็น Stable Sort และไม่ต้องใช้หน่วยความจำเพิ่มเติม
    
*   **ความซับซ้อน**: $O(n^{2})$ ไม่เหมาะกับชุดข้อมูลขนาดใหญ่
    

#### 2.3 Insertion Sort

*   **วิธีการ**: นำข้อมูลแต่ละตัวจากส่วนที่ยังไม่เรียงมา "แทรก" ในตำแหน่งที่ถูกต้องของส่วนที่เรียงแล้ว (คล้ายการเรียงไพ่ในมือ)
    
*   **ประสิทธิภาพ**: ดีที่สุดเมื่อข้อมูลเรียงอยู่แล้ว ($O(n)$) และเป็น Stable Sort
    

#### 2.4 Merge Sort

*   **วิธีการ**: ใช้หลักการ **Divide and Conquer** โดยแบ่งอาเรย์ออกเป็นครึ่งๆ จนเหลือขนาด 1 แล้วจึง Merge กลับเข้าด้วยกันแบบเรียงลำดับ
    
*   **ความซับซ้อน**: $O(n\\log n)$ เสมอในทุกกรณี
    
*   **ข้อเสีย**: ต้องการพื้นที่เพิ่มเติม $O(n)$ สำหรับการ Merge
    

### 2.5 Quick Sort

*   **วิธีการ**: เลือก **Pivot** แล้วจัดกลุ่มข้อมูล (Partition) ให้ฝั่งซ้ายน้อยกว่า Pivot และฝั่งขวามากกว่า Pivot จากนั้นทำซ้ำแบบ Recursive
    
*   **ความซับซ้อน**: เฉลี่ย $O(n \\log n)$ แต่กรณีแย่ที่สุดคือ $O(n^{2})$ (หากเลือก Pivot ไม่ดี)
    
*   **ข้อดี**: ทำงานรวดเร็วและประหยัดหน่วยความจำ (Cache Friendly)
    

### 3\. ตัวอย่างโค้ดรวม (Comprehensive C++ Code)

```cpp
#include <bits/stdc++.h> // รวมไลบรารีพื้นฐานทั้งหมด 
using namespace std;

// ฟังก์ชันพิมพ์ข้อมูล (Vector) 
void printVector(const vector<int>& arr) {
    for (int num : arr) cout << num << " ";
    cout << endl;
}

// 1. Selection Sort 
void selectionSort(vector<int> &arr) {
    int n = arr.size();
    for (int i = 0; i < n - 1; ++i) {
        int min_idx = i;
        for (int j = i + 1; j < n; ++j)
            if (arr[j] < arr[min_idx]) min_idx = j;
        swap(arr[i], arr[min_idx]); // สลับตำแหน่งที่ถูกต้อง 
    }
}

// 2. Bubble Sort (Optimized) 
void bubbleSort(vector<int>& arr) {
    int n = arr.size();
    bool swapped;
    for (int i = 0; i < n - 1; i++) {
        swapped = false;
        for (int j = 0; j < n - i - 1; j++) {
            if (arr[j] > arr[j + 1]) {
                swap(arr[j], arr[j + 1]);
                swapped = true;
            }
        }
        if (!swapped) break; // ถ้าไม่มีการสลับเลยให้หยุด 
    }
}

// 3. Insertion Sort 
void insertionSort(vector<int>& arr) {
    int n = arr.size();
    for (int i = 1; i < n; ++i) {
        int key = arr[i], j = i - 1;
        while (j >= 0 && arr[j] > key) { // เลื่อนตำแหน่งเพื่อหาที่แทรก 
            arr[j + 1] = arr[j];
            j = j - 1;
        }
        arr[j + 1] = key;
    }
}

// 4. Merge Sort (Helper & Main) 
void merge(vector<int>& arr, int left, int mid, int right) {
    int n1 = mid - left + 1, n2 = right - mid;
    vector<int> L(n1), R(n2);
    for (int i = 0; i < n1; i++) L[i] = arr[left + i];
    for (int j = 0; j < n2; j++) R[j] = arr[mid + 1 + j];
    int i = 0, j = 0, k = left;
    while (i < n1 && j < n2) {
        if (L[i] <= R[j]) arr[k++] = L[i++];
        else arr[k++] = R[j++];
    }
    while (i < n1) arr[k++] = L[i++];
    while (j < n2) arr[k++] = R[j++];
}

void mergeSort(vector<int>& arr, int left, int right) {
    if (left >= right) return;
    int mid = left + (right - left) / 2;
    mergeSort(arr, left, mid);
    mergeSort(arr, mid + 1, right);
    merge(arr, left, mid, right);
}

// 5. Quick Sort (Partition & Main) 
int partition(vector<int>& arr, int low, int high) {
    int pivot = arr[high], i = low - 1;
    for (int j = low; j <= high - 1; j++) {
        if (arr[j] < pivot) swap(arr[++i], arr[j]);
    }
    swap(arr[i + 1], arr[high]);
    return i + 1;
}

void quickSort(vector<int>& arr, int low, int high) {
    if (low < high) {
        int pi = partition(arr, low, high);
        quickSort(arr, low, pi - 1);
        quickSort(arr, pi + 1, high);
    }
}

int main() {
    vector<int> arr = {64, 34, 25, 12, 22, 11, 90};

    // ทดลองใช้ QuickSort เป็นตัวอย่าง
```
